EDG_LOT_disaster_frank_1_GER_EFFECT = {
	every_province = {
		limit = {
			or = {
				region = south_german_region
				region = north_german_region
			}
			not = { culture = root }
			not = { area = lower_rhineland_area }
		}
		add_core = GER
		if = {
			limit = { exists = GER }
			cede_province = GER
		}
		else = {
			cede_province = ROOT
		}
	}
	if = {
		limit = { exists = GER }
	}
	else = {
		release = GER
	}
	GER = {
		add_government_reform = EDG_LOT_the_frank_war_monarchy
		every_owned_province = {
			limit = { not = { is_core = root } }
			GER = {
				add_government_power = {
					mechanic_type = EDG_LOT_the_frank_war_monarchy_ability
					power_type = EDG_LOT_the_frank_war_monarchy_power
					value = 1
				}
			}
		}
		set_in_empire = no
		add_stability = 3
		set_government_rank = 3
		add_adm_power = 600
		add_dip_power = 600
		add_treasury = 5000
		add_manpower = 300
		add_prestige = 100
		add_prestige = 100
		define_ruler = {
			name = "伯恩哈德 I 「阿喀琉斯」"
			dynasty = "冯 维特尔斯巴赫"
			adm = 2
			dip = 5
			mil = 6
			age = 45
			claim = 95
		}
		define_ruler_to_general = {
			fire = 5
			shock = 5
			manuever = 4
			siege = 0
			death_date = 1590.11.24
		}
		for = {
			amount = 200
			effect = "
				capital_scope = {
					infantry = GER
				}
			"
		}
	}

      declare_war_with_cb = {
        who = GER
        casus_belli = cb_EDG_LOT_drank_war
      }
}
